Accès hors ligne / Stockage : Web Storage 

<textarea id="storage-text-area" rows="8" cols="60"></textarea>
<button id="localstorage-save">Save !</button>

<script>
	// LocalStorage => persistent storage
	// SessionStorage => per tab storage
	var saveButton = document.getElementById('localstorage-save');
	var storageTextArea = document.getElementById('storage-text-area');
	saveButton.addEventListener('click', function () {
	  window.localStorage.setItem('gdays', storageTextArea.value);
	  window.localStorage.setItem('timestamp', (new Date()).getTime());
	}, false);
	storageTextArea.value = window.localStorage.getItem('gdays');
</script>
