Dessin d'une image sur une Canvas

http://gdays-html5.appspot.com/gmaroc-codelab/images/google-logo-small.png

<canvas id="monTroisiemeCanvas" width="200" height="100" style="border:1px solid #c3c3c3;">
	Your browser does not support the canvas element.
</canvas>

<script>
	var c=document.getElementById("monTroisiemeCanvas");
	var ctx=c.getContext("2d");
	var img=new Image();
	img.onload = function(){
		ctx.drawImage(img,0,0);
	};
	img.src="gmaroc.png";
</script>

CSS3

Exemple d'animation

<style type="text/css"> 
	#monSecondDiv
	{
		width: 300px;
		height: 300px;
		background: red;
		animation: monanimation 5s;
		-webkit-animation: monanimation 5s; /* Safari and Chrome */
	}
	
	@-webkit-keyframes monanimation /* Safari and Chrome */
	{
		0%   {background:red; width:300px;}
		25%  {background:yellow; width:200px;}
		50%  {background:blue; width:250px;}
		100% {background:green; width:320px;}
	}
</style>

<div id="monSecondDiv"></div>

Localstorage 

<script>
	function clickCounter()
	{
	if(typeof(Storage)!=="undefined")
	  {
	  if (localStorage.clickcount)
	    {
	    localStorage.clickcount=Number(localStorage.clickcount)+1;
	    }
	  else
	    {
	    localStorage.clickcount=1;
	    }
	  document.getElementById("result").innerHTML="Vous avez clique " + 
	  		localStorage.clickcount + " fois.";
	  }
	else
	  {
	  document.getElementById("result").innerHTML="Indisponible !";
	  }
	}
</script>

<button onclick="clickCounter()" type="button">Appuyez !</button>
<div id="result"></div>
